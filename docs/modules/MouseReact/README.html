MouseReact モジュール

MouseReactモジュールは、マウス入力に反応するExpresserComponentを提供するモジュールです。マウスカーソルの位置追従や、マウス操作によるインタラクションを簡単に実装できます。

概要

このモジュールは以下の機能を提供します：

  • マウス位置追従: オブジェクトをマウスカーソルに追従させる
  • リアルタイム更新: マウス移動に即座に反応
  • 自動依存関係: 必要なコンポーネントを自動で取得
  • シンプルなインテグレーション: 簡単にオブジェクトに追加可能

主要機能

FollowMousePosition クラス

マウスカーソルの位置を追従するExpresserComponentです。

[VaNillaModuleVersion(0, 0, 1, 0)]
[RequiredExpresserComponent(typeof(HasPosition), auto_member_generate: true)]
[RequiredExpresserComponent(typeof(MouseInput.MouseInputComponent), auto_member_generate: true)]
public partial class FollowMousePostion : IExpresserComponent
{
    // HasPositionとMouseInputComponentは自動生成される
    
    public bool SetUp(Object object__) => false;
    public void TearDown(Object object__) { }
    
    public bool Execute(Object object__, in Time current_time)
    {
        // マウス位置をオブジェクトの位置に設定
        hasPosition.Position = MouseInput.MouseInputComponent.GetCursor();
        return false;
    }
    
    public bool Active { get; set; }
    public Execute.ExecuteType ExecuteType { get; set; }
}

条件付きマウス追従

public class ConditionalMouseFollower : IExpresserComponent
{
    private HasPosition position;
    private MouseInput.MouseInputComponent mouseInput;
    
    public bool ShouldFollow { get; set; } = true;
    public Func<bool> FollowCondition { get; set; }
    
    public bool SetUp(Object object__)
    {
        position = object__.GetExpresserComponent<HasPosition>();
        mouseInput = object__.GetExpresserComponent<MouseInput.MouseInputComponent>();
        return true;
    }
    
    public bool Execute(Object object__, in Time current_time)
    {
        bool shouldFollow = ShouldFollow;
        if (FollowCondition != null)
        {
            shouldFollow &= FollowCondition();
        }
        
        if (shouldFollow)
        {
            position.Position = MouseInput.MouseInputComponent.GetCursor();
        }
        
        return true;
    }
    
    public void TearDown(Object object__) { }
    public bool Active { get; set; } = true;
    public Execute.ExecuteType ExecuteType { get; set; } = Execute.ExecuteType.Update;
}

// 使用例
var conditionalFollower = ObjectActivator.Activate<MouseFollowObject>();
var component = conditionalFollower.GetExpresserComponent<ConditionalMouseFollower>();

// 特定の条件でのみ追従
component.FollowCondition = () => 
{
    // マウスが画面の上半分にある時のみ追従
    var mousePos = MouseInput.MouseInputComponent.GetCursor();
    return mousePos.Y < ScreenHeight / 2;
};

スムーズマウス追従

public class SmoothMouseFollower : IExpresserComponent
{
    private HasPosition position;
    private MouseInput.MouseInputComponent mouseInput;
    
    public float FollowSpeed { get; set; } = 5.0f;
    public float StopDistance { get; set; } = 1.0f;
    
    public bool SetUp(Object object__)
    {
        position = object__.GetExpresserComponent<HasPosition>();
        mouseInput = object__.GetExpresserComponent<MouseInput.MouseInputComponent>();
        return true;
    }
    
    public bool Execute(Object object__, in Time current_time)
    {
        var targetPos = MouseInput.MouseInputComponent.GetCursor();
        var currentPos = position.Position;
        
        var distance = Vector2.Distance(currentPos, targetPos);
        
        if (distance > StopDistance)
        {
            var direction = Vector2.Normalize(targetPos - currentPos);
            var moveDistance = FollowSpeed * current_time.DeltaTime;
            
            if (moveDistance > distance)
            {
                position.Position = targetPos;
            }
            else
            {
                position.Position = currentPos + direction * moveDistance;
            }
        }
        
        return true;
    }
    
    public void TearDown(Object object__) { }
    public bool Active { get; set; } = true;
    public Execute.ExecuteType ExecuteType { get; set; } = Execute.ExecuteType.Update;
}

    public override bool SetUp()
    {
        return true;
    }
}

public class DragBehavior : IExpresserComponent
{
    private HasPosition position;
    private MouseInput.MouseInputComponent mouseInput;
    private HasCollision collision;
    
    private bool isDragging = false;
    private Vector2 dragOffset;
    
    public bool SetUp(Object object__)
    {
        position = object__.GetExpresserComponent<HasPosition>();
        mouseInput = object__.GetExpresserComponent<MouseInput.MouseInputComponent>();
        collision = object__.GetExpresserComponent<HasCollision>();
        
        collision.OnCollisionEnter += OnMouseClick;
        return true;
    }
    
    public bool Execute(Object object__, in Time current_time)
    {
        if (isDragging)
        {
            position.Position = MouseInput.MouseInputComponent.GetCursor() + dragOffset;
            
            // マウスボタンが離された場合
            if (!MouseInput.MouseInputComponent.Down(MouseInput.Button.Left))
            {
                isDragging = false;
            }
        }
        
        return true;
    }
    
    private void OnMouseClick(CollisionResult result)
    {
        if (MouseInput.MouseInputComponent.Down(MouseInput.Button.Left))
        {
            isDragging = true;
            dragOffset = position.Position - MouseInput.MouseInputComponent.GetCursor();
        }
    }
    
    public void TearDown(Object object__) { }
    public bool Active { get; set; } = true;
    public Execute.ExecuteType ExecuteType { get; set; } = Execute.ExecuteType.Update;
}

マウス位置に基づくパーティクル生成

public class MouseTrailEffect : IExpresserComponent
{
    private MouseInput.MouseInputComponent mouseInput;
    private Vector2 lastMousePosition;
    
    public float ParticleSpawnRate { get; set; } = 0.1f;
    private float timeSinceLastSpawn = 0.0f;
    
    public bool SetUp(Object object__)
    {
        mouseInput = object__.GetExpresserComponent<MouseInput.MouseInputComponent>();
        lastMousePosition = MouseInput.MouseInputComponent.GetCursor();
        return true;
    }
    
    public bool Execute(Object object__, in Time current_time)
    {
        var currentMousePos = MouseInput.MouseInputComponent.GetCursor();
        
        // マウスが移動している場合
        if (Vector2.Distance(currentMousePos, lastMousePosition) > 1.0f)
        {
            timeSinceLastSpawn += current_time.DeltaTime;
            
            if (timeSinceLastSpawn >= ParticleSpawnRate)
            {
                SpawnTrailParticle(currentMousePos);
                timeSinceLastSpawn = 0.0f;
            }
        }
        
        lastMousePosition = currentMousePos;
        return true;
    }
    
    private void SpawnTrailParticle(Vector2 position)
    {
        var particle = ObjectActivator.Activate<TrailParticle>();
        particle.GetExpresserComponent<HasPosition>().Position = position;
    }
    
    public void TearDown(Object object__) { }
    public bool Active { get; set; } = true;
    public Execute.ExecuteType ExecuteType { get; set; } = Execute.ExecuteType.Update;
}

パフォーマンス考慮事項

  • 軽量実装: 最小限の計算でマウス追従を実現
  • 効率的な更新: 必要な時のみ位置更新
  • メモリ効率: 静的な依存関係で最小限のメモリ使用

注意事項

依存関係

  • HasPosition: オブジェクトの位置情報が必要
  • MouseInput.MouseInputComponent: マウス入力情報が必要
  • これらのコンポーネントはauto_member_generate: trueにより自動生成される

実行順序

  • マウス入力の取得後に位置更新が行われる
  • 他の位置依存コンポーネントとの実行順序に注意

依存関係

  • Transform - Vector2、Point2型
  • MouseInput - マウス入力システム
  • ExpresserComponent - コンポーネントシステム

関連モジュール

  • MouseInput - マウス入力処理
  • Collision - マウスクリック判定
  • Texture - マウス追従オブジェクトの描画
  • UI - UIエレメントとの組み合わせ

このモジュールにより、マウス操作に反応するインタラクティブなオブジェクトを簡単に実装できます。
