STEP 2: プレイヤーの移動を実装しよう

このステップでは、キーボード入力を使ってプレイヤーキャラクターを動かす方法を学びます。

🎯 このステップで学ぶこと

  • キーボード入力の取得方法
  • KeyConfigシステムの使い方

  • 画面端での移動制限

🎮 操作方法

💻 コードの実装

1. Player.cs - 移動機能の追加

src/Player.csを更新します：

using VaNilla;
using VaNilla.Render;
using VaNilla.Transform;
using VaNilla.Texture;
using VaNilla.KeyConfig;
using VaNilla.Keyboard;

namespace Tutorial
{
    /// <summary>
    /// プレイヤー（自機）クラス
    /// STEP2: 移動機能を追加
    /// </summary>
    public class Player : BasicObject, HasPosition, HasTexture
    {
        // === HasPositionインターフェースの実装 ===
        public Point2 Position { get; set; }
        
        // === HasTextureインターフェースの実装 ===
        private TextureResource texture;
        public TextureResource Texture 
        { 
            get => texture;
            set => texture = value;
        }
        
        // === 移動関連の定数 ===
        private const float MOVE_SPEED = 200.0f;      // 通常移動速度（ピクセル/秒）
        private const float SLOW_MOVE_SPEED = 80.0f;  // 低速移動速度（ピクセル/秒）
        private const int PLAYER_SIZE = 32;           // プレイヤーのサイズ
        
        public override void Setup()
        {
            Position = new Point2(320, 400);
            Texture = TextureResourceManager.GetTextureResource(TextureResourceType.Player);
            
            Logger.Log("プレイヤーを作成: 矢印キーで移動、Shiftで低速移動");
        }
        
        public override void Update()
        {
            HandleMovement();
        }
        
        /// <summary>
        /// 移動処理
        /// </summary>
        private void HandleMovement()
        {
            // 現在の移動速度を決定（Shiftキーで低速）
            float currentSpeed = MOVE_SPEED;
            if (KeyboardInput.IsPushed(KeyCode.LeftShift) || 
                KeyboardInput.IsPushed(KeyCode.RightShift))
            {
                currentSpeed = SLOW_MOVE_SPEED;
            }
            
            // フレームレート非依存の移動量を計算
            // TimeManager.DeltaTimeは前フレームからの経過時間（秒）
            float moveAmount = currentSpeed * TimeManager.DeltaTime;
            
            // 移動ベクトルを初期化
            float dx = 0;
            float dy = 0;
            
            // キー入力に応じて移動方向を決定
            if (KeyboardInput.IsPushed(KeyCode.Left))
            {
                dx = -moveAmount;  // 左に移動
            }
            else if (KeyboardInput.IsPushed(KeyCode.Right))
            {
                dx = moveAmount;   // 右に移動
            }
            
            if (KeyboardInput.IsPushed(KeyCode.Up))
            {
                dy = -moveAmount;  // 上に移動
            }
            else if (KeyboardInput.IsPushed(KeyCode.Down))
            {
                dy = moveAmount;   // 下に移動
            }
            
            // 斜め移動の速度調整（√2で割って正規化）
            if (dx != 0 && dy != 0)
            {
                dx *= 0.707f;  // 1/√2 ≈ 0.707
                dy *= 0.707f;
            }
            
            // 新しい位置を計算
            float newX = Position.X + dx;
            float newY = Position.Y + dy;
            
            // 画面端での移動制限
            newX = MathUtil.Clamp(newX, PLAYER_SIZE / 2, 640 - PLAYER_SIZE / 2);
            newY = MathUtil.Clamp(newY, PLAYER_SIZE / 2, 480 - PLAYER_SIZE / 2);
            
            // 位置を更新
            Position = new Point2((int)newX, (int)newY);
        }
        
        public override void Draw()
        {
            BasicRenderer.Render(this);
        }
    }
}

2. Game.cs - デバッグ情報の追加

src/Game.csを更新します：

using VaNilla;
using VaNilla.Renderer;
using VaNilla.Keyboard;

namespace Tutorial
{
    /// <summary>
    /// ゲームのメインクラス
    /// STEP2: 移動情報の表示を追加
    /// </summary>
    public class Game : BasicObject
    {
        private Player player;
        
        public override void Setup()
        {
            Logger.Log("STEP2: プレイヤーの移動");
            player = ObjectActivator.Activate<Player>();
        }

        public override void Update()
        {
            // 画面をクリア
            Utils.ClearScreen((20, 20, 60));
            
            // ESCキーで終了（デバッグ用）
            if (KeyboardInput.IsPushedMoment(KeyCode.Escape))
            {
                Logger.Log("ESCキーが押されました。終了します。");
                SupremeArbiter.Shutdown();
            }
        }

        public override void Draw()
        {
            // タイトル
            Utils.DrawString(
                "STEP 2: Player Movement", 
                10, 10,
                (255, 255, 255)
            );
            
            // 操作説明
            Utils.DrawString(
                "Arrow Keys: Move | Shift: Slow | ESC: Exit",
                10, 30,
                (200, 200, 200)
            );
            
            // プレイヤー情報
            if (player != null)
            {
                Utils.DrawString(
                    $"Position: ({player.Position.X}, {player.Position.Y})",
                    10, 50,
                    (255, 255, 0)
                );
                
                // 移動状態の表示
                string moveState = "STOP";
                if (KeyboardInput.IsPushed(KeyCode.LeftShift) || 
                    KeyboardInput.IsPushed(KeyCode.RightShift))
                {
                    moveState = "SLOW";
                }
                else if (KeyboardInput.IsPushed(KeyCode.Left) || 
                         KeyboardInput.IsPushed(KeyCode.Right) ||
                         KeyboardInput.IsPushed(KeyCode.Up) || 
                         KeyboardInput.IsPushed(KeyCode.Down))
                {
                    moveState = "MOVE";
                }
                
                Utils.DrawString(
                    $"State: {moveState}",
                    10, 70,
                    moveState == "STOP" ? (100, 100, 100) : 
                    moveState == "SLOW" ? (100, 200, 255) : (255, 100, 100)
                );
            }
            
            // フレームレート表示
            Utils.DrawString(
                $"FPS: {FpsCalculator.CurrentFps:F1}",
                540, 10,
                (0, 255, 0)
            );
        }
    }
}

🔑 重要な概念

KeyboardInput

  • IsPushed(): キーが押されている間true
  • IsPushedMoment(): キーが押された瞬間だけtrue
  • IsReleasedMoment(): キーが離された瞬間だけtrue

TimeManager.DeltaTime

  • 前フレームからの経過時間（秒）
  • フレームレート非依存の動作を実現
  • 60FPSなら約0.0167秒

MathUtil.Clamp

  • 値を指定範囲内に制限
  • Clamp(value, min, max)

斜め移動の正規化

  • X方向とY方向の両方に移動する時、速度が√2倍にならないよう調整
  • 0.707（≈ 1/√2）を掛けることで一定速度を保つ
