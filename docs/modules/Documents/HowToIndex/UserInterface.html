HPゲージの実装

線形HPバー

public void DrawHPBar(int x, int y, int current, int max)
{
    int barWidth = 200;
    int barHeight = 20;
    
    // 背景（黒）
    Utils.FillRectangle(x, y, barWidth, barHeight, (0, 0, 0));
    
    // HPバー（現在値に応じて赤→黄→緑）
    float ratio = (float)current / max;
    int fillWidth = (int)(barWidth * ratio);
    
    var color = ratio > 0.6f ? (0, 255, 0) :    // 緑
                ratio > 0.3f ? (255, 255, 0) :  // 黄
                (255, 0, 0);                     // 赤
    
    Utils.FillRectangle(x, y, fillWidth, barHeight, color);
    
    // 枠線
    Utils.DrawRectangle(x, y, barWidth, barHeight, (255, 255, 255));
}

キーボード操作

メニューナビゲーション

public class MenuNavigationComponent : IExpresserComponent
{
    private int selectedIndex = 0;
    private string[] menuItems = {"スタート", "オプション", "終了"};
    
    public bool Execute(Object obj, in Time.Time time)
    {
        if (KeyboardInput.IsPushedMoment(KeyCode.Up))
            selectedIndex = Math.Max(0, selectedIndex - 1);
        
        if (KeyboardInput.IsPushedMoment(KeyCode.Down))
            selectedIndex = Math.Min(menuItems.Length - 1, selectedIndex + 1);
        
        if (KeyboardInput.IsPushedMoment(KeyCode.Return))
            OnMenuItemSelected(selectedIndex);
        
        return true;
    }
}

public class HoverEffectComponent : IExpresserComponent
{
    public bool Execute(Object obj, in Time.Time time)
    {
        var mousePos = MouseInput.GetMousePosition();
        var button = obj as Button;
        
        if (IsMouseOver(mousePos, button.bounds))
        {
            button.color = Color.Yellow;  // ハイライト
        }
        else
        {
            button.color = Color.White;   // 通常
        }
        
        return true;
    }
}

🎭 UIアニメーション

イージングを使ったアニメーション

フェードイン・アウト

public class FadeComponent : IExpresserComponent
{
    private float timer = 0;
    private float duration = 1.0f;
    
    public bool Execute(Object obj, in Time.Time time)
    {
        timer += TimeManager.DeltaTime;
        float progress = timer / duration;
        
        // イージング関数でスムーズなフェード
        float alpha = Easing.EaseOutQuad(progress, 0, 1, 1);
        
        // アルファ値を適用
        var ui = obj as UIElement;
        ui.Alpha = alpha;
        
        return progress < 1.0f;
    }
}

→ 詳細: Easing - イージング関数

スライドアニメーション

public class SlideInComponent : IExpresserComponent
{
    private Vector2 startPos;
    private Vector2 targetPos;
    private float timer = 0;
    
    public bool Execute(Object obj, in Time.Time time)
    {
        timer += TimeManager.DeltaTime;
        float progress = Math.Min(timer / 0.5f, 1.0f);
        
        // EaseOutBackでバウンス効果
        float easedProgress = Easing.EaseOutBack(progress);
        
        var currentPos = Vector2.Lerp(startPos, targetPos, easedProgress);
        obj.hasPosition.Position = new Point2((int)currentPos.X, (int)currentPos.Y);
        
        return progress < 1.0f;
    }
}

public class ButtonParticleEffect : IExpresserComponent
{
    public bool Execute(Object obj, in Time.Time time)
    {
        if (IsButtonClicked())

            // パーティクルエフェクトを生成
