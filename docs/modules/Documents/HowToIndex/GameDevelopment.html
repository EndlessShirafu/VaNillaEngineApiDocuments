ゲームの起動と初期化

最小限のゲーム起動

// Main.cs
using System;
using VaNilla;
namespace Main
{
    /// <summary>
    /// エントリーポイント
    /// VaNillaエンジンの正しい起動方法を示す
    /// </summary>
    static class Program
    {
        /// <summary>
        /// メイン関数
        /// </summary>
        static void Main(string[] args)
        {
            // SupremeArbiterの静的コンストラクタが自動実行される
            // エンジン初期化完了まで待機
            while (!SupremeArbiter.VaNillaEngineConstructed)
            {
                Thread.Sleep(1);
            }
            
            // メインゲームループ
            while (SupremeArbiter.IsRunning())
            {
                SupremeArbiter.Execute();  // 毎フレーム実行
            }
            
            // エンジン終了処理
            SupremeArbiter.End();
        }
    }
}

ゲーム固有の初期化・終了処理

using System;
using VaNilla;
using VaNilla.Debug;

namespace BootLoaderSample
{
    public class BootLoaderSampleClass
    {
        // VaNillaBootLoader 属性を付けると、VaNillaエンジンの初期化前にこのメソッドが実行される
        [VaNillaBootLoader]
        public static bool VaNillaBootLoaderMethod()
        {
            Logger.Log("VaNillaBootLoaderMethod passed");
            return true;
        }

        // GameBootLoader 属性を付けると、VaNillaエンジンの初期化直後にこのメソッドが実行される
        [GameBootLoader]
        public static bool GameBootLoaderMethod()
        {
            Logger.Log("GameBootLoaderMethod passed");
            return true;
        }

        // GameBootLoader 属性を付けると、VaNillaエンジンの終了処理直前にこのメソッドが実行される
        [GameShutdown]
        public static bool GameShutdownMethod()
        {
            Logger.Log("GameShutdownMethod passed");
            return true;
        }
    }
}

プレイヤーの移動

public class PlayerMovementComponent : IExpresserComponent
{
    public bool Execute(Object obj, in Time.Time time)
    {
        var player = obj as Player;
        if (KeyboardInput.IsPushed(KeyCode.Left))
            player.hasPosition.Position.X -= 5;
        if (KeyboardInput.IsPushed(KeyCode.Right))
            player.hasPosition.Position.X += 5;
        return true;
    }
}
