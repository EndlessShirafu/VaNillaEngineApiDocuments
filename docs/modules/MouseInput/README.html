MouseInput モジュール

MouseInputモジュールは、マウス入力の検出・管理機能を提供するモジュールです。ボタン状態の取得、座標情報の管理、ホイール操作、カーソル制御などの包括的なマウス機能をサポートします。

概要

このモジュールは以下の機能を提供します：

  • マウスボタン入力: 左・中・右ボタンの押下状態検出
  • 座標管理: カーソル位置の取得・設定
  • ホイール操作: 縦横スクロールホイールの検出
  • カーソル制御: 表示・非表示、移動制限、速度調整
  • ExpresserComponent統合: オブジェクト指向のマウス入力処理

APIリファレンス

詳細なAPIドキュメントはVaNilla MouseInput APIリファレンスをご覧ください。

対応プラットフォーム

  • ✅ DxLib: フル機能実装済
  • ⚠️ Unity: 開発途中。Unity Input System統合

主要コンポーネント

MouseInput クラス（静的API）

システム全体のマウス入力を管理する静的クラスです。

public static class MouseInput
{
    // 初期化・更新
    public static void Initialize();
    public static void UpdateMouseState();
    
    // ボタン状態取得
    public static bool JustDown(Button btn);    // 現フレームで押された
    public static bool Down(Button btn);        // 押されている
    public static bool JustUp(Button btn);      // 現フレームで離された
    
    // 座標取得・設定
    public static float GetCursorX();
    public static float GetCursorY();
    public static Point2 GetCursor();
    public static void SetCursorX(float x);
    public static void SetCursorY(float y);
    public static void SetCursor(Point2 point);
    
    // ホイール
    public static float GetWheelScroll();           // 縦スクロール
    public static float GetHorizontalWheelScroll(); // 横スクロール
    
    // カーソル制御
    public static void SetForceCursorInnerWindow(bool flag);
    public static bool GetForceCursorInnerWindow();
    public static void SetMouseDispFlag(bool flag);
    public static bool GetMouseDispFlag();
    public static bool SetMouseCursorSpeed(float speed);
    public static float GetMouseCursorSpeed();
}

MouseInputComponent（ExpresserComponent）

オブジェクト指向のマウス入力処理を提供するコンポーネントです。

[VaNillaModuleVersion(0, 0, 1, 0)]
public class MouseInputComponent : IExpresserComponent
{
    // 静的APIのラッパー関数
    public static bool JustDown(Button btn);
    public static bool Down(Button btn);
    public static bool JustUp(Button btn);
    public static float GetCursorX();
    public static float GetCursorY();
    public static Point2 GetCursor();
}

使用方法

基本的な使用例

// ゲーム初期化時（一度のみ）
MouseInput.Initialize();

// 毎フレーム実行
MouseInput.UpdateMouseState();

// ボタン状態の確認
if (MouseInput.JustDown(Button.Left))
{
    Logger.Log("左ボタンが押された");
}

if (MouseInput.Down(Button.Right))
{
    Logger.Log("右ボタンが押されている");
}

if (MouseInput.JustUp(Button.Middle))
{
    Logger.Log("中央ボタンが離された");
}

// 座標の取得
float mouseX = MouseInput.GetCursorX();
float mouseY = MouseInput.GetCursorY();
Point2 mousePos = MouseInput.GetCursor();

Logger.Log($"マウス座標: ({mouseX}, {mouseY})");

// ホイール操作の検出
float wheelScroll = MouseInput.GetWheelScroll();
if (wheelScroll > 0)
{
    Logger.Log("ホイールを奥に回した");
}
else if (wheelScroll < 0)
{
    Logger.Log("ホイールを手前に回した");
}

ExpresserComponentとしての使用

[AutoGenerateConstructor]
[DefaultExpresserComponent(typeof(MouseInputComponent))]
[DefaultExpresserComponent(typeof(HasPosition))]
[DefaultExpresserComponent(typeof(HasClickHandler))]
public partial class ClickableObject : Object
{
    public override bool SetUp()
    {
        return true;
    }
}

[RequiredExpresserComponent(typeof(MouseInputComponent))]
[RequiredExpresserComponent(typeof(HasPosition))]
public class HasClickHandler : IExpresserComponent
{
    public bool Execute(Object object__, in Time current_time)
    {
        if (MouseInputComponent.JustDown(Button.Left))
        {
            var position = object__.GetExpresserComponent<HasPosition>();
            var mousePos = MouseInputComponent.GetCursor();
            
            if (IsPointInObject(mousePos, position))
            {
                OnClicked();
            }
        }
        return true;
    }
    
    private void OnClicked()
    {
        Logger.Log("オブジェクトがクリックされた");
    }
}

ドラッグ&ドロップの実装例

public class DraggableComponent : IExpresserComponent
{
    private bool isDragging = false;
    private Point2 dragOffset;
    
    public bool Execute(Object object__, in Time current_time)
    {
        var position = object__.GetExpresserComponent<HasPosition>();
        var mousePos = MouseInputComponent.GetCursor();
        
        if (MouseInputComponent.JustDown(Button.Left) && IsPointInObject(mousePos, position))
        {
            isDragging = true;
            dragOffset = new Point2(
                mousePos.x - position.X,
                mousePos.y - position.Y
            );
        }
        
        if (isDragging)
        {
            if (MouseInputComponent.Down(Button.Left))
            {
                // ドラッグ中の移動
                position.X = mousePos.x - dragOffset.x;
                position.Y = mousePos.y - dragOffset.y;
            }
            else
            {
                // ドラッグ終了
                isDragging = false;
            }
        }
        
        return true;
    }
}

マウスカーソル制御

// カーソルを非表示にする
MouseInput.SetMouseDispFlag(false);

// カーソルをウィンドウ内に制限
MouseInput.SetForceCursorInnerWindow(true);

// マウス感度を調整
MouseInput.SetMouseCursorSpeed(0.5f); // 半分の速度

// 特定の位置にカーソルを設定
MouseInput.SetCursor(new Point2(100, 50));

ホイール操作の詳細例

// 縦スクロール（通常のホイール）
float verticalScroll = MouseInput.GetWheelScroll();
if (Math.Abs(verticalScroll) >= 1.0f)
{
    if (verticalScroll > 0)
    {
        ZoomIn(verticalScroll);
    }
    else
    {
        ZoomOut(-verticalScroll);
    }
}

// 横スクロール（チルトホイール）
float horizontalScroll = MouseInput.GetHorizontalWheelScroll();
if (Math.Abs(horizontalScroll) >= 1.0f)
{
    ScrollHorizontally(horizontalScroll);
}

注意事項

フレーム管理

  • UpdateMouseState()は毎フレーム最初に呼び出す必要があります
  • JustDown/JustUpは1フレームのみtrueを返します

座標系

  • 座標はウィンドウのクライアント領域基準（左上が原点）
  • DxLibとUnityで座標系の違いに注意

パフォーマンス

  • マウス状態の更新は軽量ですが、座標取得は必要な時のみ行うことを推奨
  • ホイール値は現フレームでの変化量（累積値ではない）

プラットフォーム固有の特徴

DxLib実装

  • Windows APIを直接使用
  • 高精度なマウス入力対応
  • ウィンドウフォーカス状態の自動管理

Unity実装

  • Unity Input Systemとの統合
  • マルチプラットフォーム対応
  • UnityのUIシステムとの協調

依存関係

  • Transform - Point2型の座標表現
  • TimeManager - フレーム同期
  • ExpresserComponent - コンポーネントシステム

関連モジュール

  • ClickableRegion - クリック可能領域管理
  • MouseReact - マウス反応コンポーネント
  • PlayerInput - 抽象化された入力管理

拡張可能性

  • マルチマウス対応: 複数マウスデバイスの同時サポート
  • ジェスチャー認識: スワイプ、ピンチなどのマウスジェスチャー
  • カーソル画像変更: 状況に応じたカーソル画像の切り替え
  • 入力履歴: マウス操作の記録・再生機能
